#include <WiFi.h>
#include "ThingSpeak.h"
#include <Wire.h>
#include <DHT.h>
#include <MPU6050.h>

//  WiFi Credentials
const char* ssid = "AM-128";
const char* password = "password";

//  ThingSpeak
unsigned long channelID = 3339984;
const char* apiKey = "757U2DYA9T3EKGF4";
WiFiClient client;

//  Pins
const int ir1 = 25;
const int ir2 = 26;
const int ir3 = 27;
const int currentPin = 34;

const int relayPin = 23;
const int ledPin = 19;
const int buzzerPin = 18;

#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

MPU6050 mpu;

//  Variables
int queue = 0;
int prevIR1 = HIGH, prevIR2 = HIGH, prevIR3 = HIGH;

unsigned long startTime = 0;
unsigned long processTime = 0;
bool processRunning = false;

unsigned long lastUpdate = 0;

//  Thresholds
const float tempThreshold = 35.0;
const float vibrationThreshold = 1.5;
const float currentThreshold = 1.02;

//  ACS712
float sensitivity = 0.185;
float offsetVoltage = 1.52;

//  ML constants
float T_crit = 35.0;
float V_crit = 1.5;
float Q_crit = 3.0;
float I_crit = 1.3;
float tau_crit = 50.0;

float w_T = 0.3, w_V = 0.25, w_I = 0.2, w_tau = 0.15, w_Q = 0.1;

float alpha = 0.5;
float beta = 0.3;
float gamma_coeff = 0.2;

float R_max = 100.0;

//  Motor state
bool motorRunning = true;

// ================================
//  HEALTH REPORT FUNCTION
// ================================
void printHealthMetrics(float D, float RUL, float PMI, String decision) {
  Serial.println("========= HEALTH REPORT =========");

  Serial.print("Degradation Score (D): ");
  Serial.println(D);

  Serial.print("Remaining Useful Life (RUL): ");
  Serial.print(RUL);
  Serial.println(" %");

  Serial.print("Predictive Maintenance Index (PMI): ");
  Serial.println(PMI);

  Serial.print("System Decision: ");
  Serial.println(decision);

  Serial.println("=================================");
}

void setup() {
  Serial.begin(115200);

  pinMode(ir1, INPUT);
  pinMode(ir2, INPUT);
  pinMode(ir3, INPUT);

  pinMode(relayPin, OUTPUT);
  pinMode(ledPin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);

  digitalWrite(relayPin, HIGH); // Motor ON

  dht.begin();
  Wire.begin(21, 22);
  mpu.initialize();

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected");

  ThingSpeak.begin(client);
}

void loop() {

  // ================= IR SYSTEM =================
  int currIR1 = digitalRead(ir1);
  int currIR2 = digitalRead(ir2);
  int currIR3 = digitalRead(ir3);

  if (prevIR1 == HIGH && currIR1 == LOW) queue++;

  if (prevIR2 == HIGH && currIR2 == LOW) {
    if (queue > 0) queue--;
    startTime = millis();
    processRunning = true;
  }

  if (prevIR3 == HIGH && currIR3 == LOW) {
    if (processRunning) {
      processTime = millis() - startTime;
      processRunning = false;
    }
  }

  prevIR1 = currIR1;
  prevIR2 = currIR2;
  prevIR3 = currIR3;

  // ================= SENSOR DATA =================
  float tempC = dht.readTemperature();
  if (isnan(tempC)) tempC = 0;

  int16_t ax, ay, az;
  mpu.getAcceleration(&ax, &ay, &az);

  float vibration = sqrt(
    (ax / 16384.0) * (ax / 16384.0) +
    (ay / 16384.0) * (ay / 16384.0) +
    (az / 16384.0) * (az / 16384.0)
  );

  float currentSum = 0;
  for (int i = 0; i < 10; i++) {
    int raw = analogRead(currentPin);
    float voltage = (raw / 4095.0) * 3.3;
    float current = (voltage - offsetVoltage) / sensitivity;
    currentSum += current;
    delay(5);
  }

  float currentAvg = currentSum / 10;
  if (currentAvg < 0) currentAvg = 0;

  float processTimeSec = processTime / 1000.0;

  // ================= ML MODEL =================
  float T_n = tempC / T_crit;
  float V_n = vibration / V_crit;
  float Q_n = queue / Q_crit;
  float I_n = currentAvg / I_crit;
  float tau_n = processTimeSec / tau_crit;

  float D = (w_T * T_n) + (w_V * V_n) + (w_I * I_n) + (w_tau * tau_n) + (w_Q * Q_n);

  float RUL = R_max * (1 - D);
  if (RUL < 0) RUL = 0;

  float A = sqrt(pow(T_n - 1, 2) + pow(V_n - 1, 2) + pow(I_n - 1, 2));

  float aging_term = 1 - (RUL / R_max);
  float PMI = (alpha * D) + (beta * A) + (gamma_coeff * aging_term);

  // ================= DECISION =================
  String decision;

  if (tempC > T_crit && vibration > V_crit) {
    decision = "STOP";
  }
  else if (tempC > T_crit) {
    decision = "TEMP_ALERT";
  }
  else if (vibration > V_crit) {
    decision = "VIB_ALERT";
  }
  else if (PMI > 0.65) {
    decision = "WARNING";
  }
  else {
    decision = "NORMAL";
  }

  // ================= MOTOR CONTROL =================
  if (decision == "STOP") {
    digitalWrite(relayPin, LOW);
    motorRunning = false;

    digitalWrite(ledPin, HIGH);
    digitalWrite(buzzerPin, HIGH);

    delay(5000);

    digitalWrite(relayPin, HIGH);
    motorRunning = true;

    digitalWrite(ledPin, LOW);
    digitalWrite(buzzerPin, LOW);
  }

  // ================= SERIAL OUTPUT =================
  printHealthMetrics(D, RUL, PMI, decision);

  Serial.print("Queue: "); Serial.print(queue);
  Serial.print(" | Temp: "); Serial.print(tempC);
  Serial.print(" | Vib: "); Serial.print(vibration);
  Serial.print(" | Current: "); Serial.print(currentAvg);
  Serial.println();

  // ================= ThingSpeak =================
  if (millis() - lastUpdate >= 15000) {
    ThingSpeak.setField(1, queue);
    ThingSpeak.setField(2, tempC);
    ThingSpeak.setField(3, vibration);
    ThingSpeak.setField(4, currentAvg);
    ThingSpeak.setField(5, processTimeSec);

    ThingSpeak.writeFields(channelID, apiKey);
    lastUpdate = millis();
  }

  Serial.println("-------------------------");
  delay(100);
}
